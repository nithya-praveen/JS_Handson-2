Q1. Calculator Write a program to make a simple calculator using switch statement that takes operator and two numbers from user and print the result after operation.

A1. const Calculator = (A, B, C) =>
{
let result;
switch(A)
{
case '+' : result = B+C;
return result;
case '-' : result = B-C;
return result;
case '*' : result = B*C;
return result;
case '/' : result = B/C;
return result;
default: return "Invalid Operator";
}
};

  • If M<=40 and M>=31 the grade will be B
  • If M<=50 and M>=41 the grade will be A
 A5. const findGrades = (a) =>
{

{

Q9. Check whether the triangle is Acute or Obtuse Write a program takes takes three angles and checks whether the triangle is acute or obtuse.

A9. const Triangle_Check = (A,B,C) =>
{
if(A<90 && B<90 && C<90)
{
return "acute";
}
else if(A<180 || B<180 || C<180)
{
return "obtuse";
}
